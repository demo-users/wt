import pandas as pd
from sklearn.preprocessing import MinMaxScaler, StandardScaler, Binarizer

data = pd.read_csv("winequality-red.csv")

# a. Rescaling using MinMaxScaler
scaler = MinMaxScaler()
data_rescaled = pd.DataFrame(scaler.fit_transform(data), columns=data.columns)
print("Rescaled Data (Min-Max Scaling):\n", data_rescaled.head(), "\n")

# b. Standardizing using StandardScaler
std_scaler = StandardScaler()
data_standardized = pd.DataFrame(std_scaler.fit_transform(data), columns=data.columns)
print("Standardized Data (Mean=0, Std=1):\n", data_standardized.head(), "\n")

# c. Binarizing using Binarizer 
binarizer = Binarizer(threshold=5)
data_binarized = pd.DataFrame(binarizer.fit_transform(data), columns=data.columns)
print("Binarized Data (Threshold=5):\n", data_binarized.head())
