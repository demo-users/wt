import pandas as pd

data = {
    'Name': ['Alice', 'Bob', 'Charlie', 'David', 'Eva'],
    'Graduation_Percentage': [85, 78, 92, 88, 76],
    'Age': [22, 23, 21, 24, 22]
}

df = pd.DataFrame(data)

average_age = df['Age'].mean()
average_percentage = df['Graduation_Percentage'].mean()

print("Students Data:\n", df)
print("\nAverage Age of Students:", average_age)
print("Average Graduation Percentage:", average_percentage)
