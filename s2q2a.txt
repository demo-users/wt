import pandas as pd

df = pd.read_csv('Data.csv')

df.loc[:, 'salary'] = df['salary'].fillna(df['salary'].mean())
df.loc[:, 'age'] = df['age'].fillna(df['age'].mean())

print("After replacing missing values:")
print(df)
