import pandas as pd
import matplotlib.pyplot as plt

df = pd.read_csv('Data.csv')

plt.plot(df['name'], df['salary'], marker='o')
plt.xlabel('Name')
plt.ylabel('Salary')
plt.title('Name vs Salary')
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()
